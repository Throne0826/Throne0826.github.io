触底：

 getcolor(){
    console.log('触发了上拉触底'),
    wx.request({
      url: 'https://www.colr.org/json/color/random',
      method:"GET",
      success:({data:res})=>{
        this.setData({
          colorlist:[...this.data.colorlist,...res.colorlist]
        })
      }
    })
  },
  onReachBottom:function(){
    this.getcolor()
  },
