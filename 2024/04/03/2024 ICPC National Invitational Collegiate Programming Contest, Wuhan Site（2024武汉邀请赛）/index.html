void solve(){
    int n,res=0;
    cin>>n;
    for(int i=1;i<=n;i++){
        int x;cin>>x;
        res+=x;
    }
    int ans=0;
    for(int i=30;i>=0;i--){
        int ver=((1ll<<i)-1)*n;
        if(res>ver)ans|=(1ll<<i),res-=min(n,res/(1ll<<i))*(1ll<<i);
    }
    cout<<ans<<"\n";
}
